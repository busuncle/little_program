can not get real i

for(var i=0; i<3; i++){
    var b = $("<input type='button' />");
    b.val("all is 3");
    b.click(function(){
        alert(i);
    });
    b.appendTo($("#not_real_i"));
}

can get real i

for(var i=0; i<3; i++){
    var b = $("<input type='button' />");
    b.val(i);
    (function(the_real_i){
        // use a trick, anonymous functions wrapper
        b.click(function(){
            alert(the_real_i);
        });
    })(i);
    b.appendTo($("#real_i"));
}
